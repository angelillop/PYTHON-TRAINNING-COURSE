#Counter (COUNT THE NUMBER OF ELEMENTS IN A LIST, STRING)
from collections import Counter

  Example:
  l = [1,1,2,1,3,1,3,2,4,5,6,5,4,5,6]
  Counter(l)
  Output:
  Counter({1: 4, 2: 2, 3: 2, 4: 2, 5: 3, 6: 2})
